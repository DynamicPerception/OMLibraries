260 void OMComHandler::_isrFire() {
261 
262  uint8_t stat;
263 
264  if( m_which )
265  stat = OMC_COM1_REG & (1 << OMC_COM1_PIN);
266  else
267  stat = OMC_COM2_REG & (1 << OMC_COM2_PIN);
268 
269  // if transitioned from high to low,
270  // record start time
271  if( ! stat ) {
272  m_isrUs = micros();
273  }
274  else {
275  // if a transition from low to high,
276  // figure out diff and call callback (if set)
277  unsigned long diff = micros() - m_isrUs;
278 
279  if( f_isrCB != 0 )
280  f_isrCB(diff);
281 
282  }
