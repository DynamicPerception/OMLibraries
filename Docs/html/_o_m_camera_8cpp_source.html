203 void OMCamera::stop() {
204 
205  // stop timer (always)
206  // we do this before bringing the pin low in case
207  // a VERY short action time was set to prevent the
208  // timer from continuing to trigger and getting stuck in
209  // a loop.
210 
211  MsTimer2::stop();
212 
213  uint8_t code = 0;
214 
215  if( m_curAct == OM_CAM_INEXP ) {
216  // expose complete
217  digitalWrite(m_shutter, LOW);
218 
219  // we do this every time, because
220  // it's possible that the flag
221  // that controls whether or not to
222  // trip focus w. shutter may have
223  // been reset during our exposure,
224  // and failing to do so would keep
225  // the focus pin high for a long
226  // time.
227 
228  digitalWrite(m_focus, LOW);
229  code = OM_CAM_EFIN;
230  }
231  else if( m_curAct == OM_CAM_INFOC ) {
232  // focus complete
233  digitalWrite(m_focus, LOW);
234  code = OM_CAM_FFIN;
235  }
236  else {
237  code = OM_CAM_WFIN;
238  }
239 
240 
241  // reset current action
242  m_curAct = OM_CAM_INCLR;
243 
244  // update camera currently engaged
245  m_isBzy = false;
246 
247 
248  if( code != 0 ) {
249  // if we got here because of stop() called from a timer, then
250  // let's send a proper signal (if callback set)
251  if( f_camSignal != 0 )
252  f_camSignal(code);
253  }
254 
255 }

279 void OMCamera::expose(unsigned long p_time) {
280 
281  // do not expose if exposure time is zero
282  if( p_time == 0 ) {
283  if( f_camSignal != 0 )
284  f_camSignal(OM_CAM_EFIN);
285 
286  return;
287  }
288  // indicate (for stop()), that we are in an exposure
289  // to properly set output states when timer2 completes
290 
291  m_curAct = OM_CAM_INEXP;
292 
293 
294  // determine if focus pin should be brought high
295  // w. the shutter pin (for some nikons, etc.)
296 
297  if( m_focusShut == true )
298  digitalWrite(m_focus, HIGH);
299 
300  digitalWrite(m_shutter, HIGH);
301 
302  // start timer to stop camera exposure
303  MsTimer2::set(p_time, OMCamera::stop);
304  MsTimer2::start();
305 
306 
307  // update camera currently engaged
308  m_isBzy = true;
309 
310 
311  // if there is a pointer to a function
312  // to be called when the camera is done,
313  // do it now
314 
315  if( f_camSignal != 0 )
316  f_camSignal(OM_CAMEXP);
317 
318  return;
319 }

343 void OMCamera::focus(unsigned int p_time) {
344 
345  // do not focus if focus time is 0
346  if( p_time == 0 ) {
347  if( f_camSignal != 0 )
348  f_camSignal(OM_CAM_FFIN);
349 
350  return;
351  }
352 
353  digitalWrite(m_focus, HIGH);
354 
355  // start timer to stop focus engage
356 
357  m_curAct = OM_CAM_INFOC;
358 
359  MsTimer2::set(p_time, OMCamera::stop);
360  MsTimer2::start();
361  // update camera currently engaged
362  m_isBzy = true;
363  // report focus in progress
364  if( f_camSignal != 0 )
365  f_camSignal(OM_CAMFOC);
366 
367 }
