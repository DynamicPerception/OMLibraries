121 uint8_t MenuContext::formatParameterText(const uint8_t idx, uint8_t* cLineBuf, uint8_t* level, uint8_t* item )
122 {
123  uint32_t value = readLiveParameter(idx);
124  *level = 0;
125  *item = 0;
126 
127  uint8_t len = 0;
128  //temporary init as
129  cLineBuf[0] = '[';
130  cLineBuf[1] = '=';
131  char* ptrStart = (char*)&cLineBuf[2];
132  if ((idx > 0) && (idx < 10)) {
133  //integer type
134  sprintf(ptrStart, "%lu", value );
135  }
136  /*dynamic list, very special case*/
137  if (idx == 10) {
138  uint32_t listIdx = value;
139  checkParamByIdx(idx, &listIdx);
140  sprintf(ptrStart, "%s", &dynList[listIdx-1][0] );
141  if (listIdx == dynListSize) {
142  *item = 2;
143  *level = CREATE_FILM_WIZARD;
144  } else {
145  *item = 1;
146  *level = AXIS_WIZARD;
147  }
148  }
149  /* magic 11(12)(13) or 14:15:16 is time type*/
150  if ((idx == 11) || (idx == 14)){
151  uint32_t tmp = value;
152  uint8_t hour = tmp/3600;
153  //tmp = tmp - ((uint32_t)hour * 3600UL); // may be
154  tmp = tmp%3600;
155  uint8_t minutes = tmp/60;
156  //tmp = tmp - ((uint32_t)minutes * 60UL); //
157  tmp = tmp%60;
158  uint8_t secs = tmp;
159  sprintf(ptrStart, "%02d:%02d:%02d", hour, minutes, secs);
160  }
161  /*magic 20,21... is list type*/
162  if ((idx >= 20) && (idx < NUMBER_OF_PARAMETERS))
163  {
164  uint32_t listIdx = value;
165  checkParamByIdx(idx, &listIdx);
166  if (idx == 20){
167  strcpy_P(ptrStart, &fixedList1[listIdx].caption[0]);
168  *item = pgm_read_byte(&fixedList1[listIdx].jumpItem);
169  *level = pgm_read_byte(&fixedList1[listIdx].jumpLevel);
170 
171  } else if (idx == 21){
172  strcpy_P(ptrStart, &fixedList2[listIdx].caption[0]);
173  *item = pgm_read_byte(&fixedList2[listIdx].jumpItem);
174  *level = pgm_read_byte(&fixedList2[listIdx].jumpLevel);
175 
176  } else if (idx == 22){
177  strcpy_P(ptrStart, &fixedList3[listIdx].caption[0]);
178  *item = pgm_read_byte(&fixedList3[listIdx].jumpItem);
179  *level = pgm_read_byte(&fixedList3[listIdx].jumpLevel);
180 
181  } else if (idx == 23){
182  strcpy_P(ptrStart, &fixedList4[listIdx][0]);
183 
184  } else if (idx == 24){
185  strcpy_P(ptrStart, &fixedList5[listIdx][0]);
186  } else {
187  //strcpy_P((char*)cLineBuf, PSTR("--noList--"));
188  }
189  }
190  len = strlen((char*)cLineBuf);
191  return len;
192 }

245  uint16_t step;
246  memcpy_P(&step, &iStepParameters[cFocusParameter], sizeof(step));
247  //check for roll over zero
248  uint32_t tmp = iModifiableValue;
249  tmp -= step*iKeyboardIncrement;
