142 uint8_t OMMoCoBus::getPacket() {
143 
144  uint8_t command_val = 0;
145  uint8_t com_byte_count = 0;
146  //uint8_t dat = 0;
147  uint8_t ret = 0;
148 
149  m_isBCast = false;
150 
151  bool not_us = false;
152 
153  // we check serial.available() here even
154  // though this->_getNextByte() does, because
155  // a timeout is applied waiting for next command
156  // in this->_getNextByte() - since this is called
157  // every cycle of the main loop, we don't want
158  // to introduce a delay unless we know of a
159  // byte waiting to be read
160 
161  int avail = m_serObj->available();
162 
163  // serial does not have data in the buffer
164  if( avail <= 0 )
165  return(0);
166 
167  // is this packet intended for us?
168  uint8_t stat = this->_targetUs();
169 
170  if(stat == OM_SER_ERR || stat == OM_SER_TIMEOUT) {
171  // got an error? timeout error, really.
172  // this->_flushSerial();
173  return(0);
174  }
175  else if( stat == OM_SER_IS_BCAST ) {
176  // command was a broadcast command
177  m_isBCast = true;
178  }
179  else if( stat != OM_SER_OK ) {
180  // command was for someone else
181  // (but we need to read the rest of the data
182  // off the line to make sure we don't keep part of it
183  // in the buffer)
184  not_us = true;
185  }
186 
187  // get command byte
188  ret = this->_getNextByte(command_val);
189 
190  if ( ret != OM_SER_OK ) {
191  // well, this is just embarassing
192  // this->_flushSerial();
193  return(0);
194  }
195 
196  if( command_val > 0 ) {
197  // get data length (even if not our own command)
198  ret = this->_getNextByte(com_byte_count);
199 
200  if ( ret != OM_SER_OK ) {
201  // well, this is just embarassing
202  // this->_flushSerial();
203  return(0);
204  }
205  }
206 
207  // check for overflow
208  // note: we will still be in error condition, as
209  // there will be values left in the serial buffer
210  // on the next read, from this command - so don't
211  // exceed it, eh? This is just to prevent memory
212  // corruption. It is not safe otherwise
213 
214  if(com_byte_count > OM_SER_BUFLEN)
215  com_byte_count = OM_SER_BUFLEN;
216 
217 
218  // clear out any previous command data
219  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
220  m_bufSize = 0;
221 
222 
223  // populate command data buffer
224  for( int i = 1; i <= com_byte_count; i++ ) {
225  // get com_byte_count character values from the serial
226  // buffer
227 
228  uint8_t dat = 0;
229  uint8_t ret = this->_getNextByte(dat);
230 
231  if( ret != OM_SER_OK )
232  return(0);
233 
234  m_serBuffer[i - 1] = dat;
235  m_bufSize++;
236  }
237 
238  if( not_us == true ) {
239  // the command was not intended for us
240  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
241  return(0);
242  }
243 
244 
245  return( command_val );
246 }

508 uint8_t OMMoCoBus::_targetUs() {
509 
510  uint8_t thsChar = 1;
511  uint8_t ret = this->_getNextByte(thsChar);
512 
513  // timeout? other error?
514  if( ret != OM_SER_OK )
515  return(OM_SER_TIMEOUT);
516 
517  // if character is null...
518  if( thsChar == 0 ) {
519  // until we get enough nulls for a start command
520  // or read something that isn't null...
521 
522  uint8_t nullCount = 0;
523 
524  while( nullCount < 5 && thsChar == 0 ) {
525  nullCount++;
526 
527  // null count met
528  if( nullCount == 5 )
529  break;
530 
531  ret = this->_getNextByte(thsChar);
532 
533  if( ret != OM_SER_OK )
534  return(OM_SER_TIMEOUT);
535 
536  }
537 
538  // did not receive a proper start sequence
539  if( nullCount < 5 )
540  return(OM_SER_ERR);
541 
542  thsChar = 0;
543 
544  ret = this->_getNextByte(thsChar);
545 
546  if( ret != OM_SER_OK )
547  return(OM_SER_TIMEOUT);
548 
549  if( thsChar != 255 )
550  return(OM_SER_ERR);
551 
552  // get target address
553 
554  unsigned int addr = 0;
555  ret = this->_getNextByte(thsChar);
556 
557  if( ret != OM_SER_OK )
558  return(OM_SER_TIMEOUT);
559 
560  addr = (unsigned int) thsChar << 8;
561 
562  ret = this->_getNextByte(thsChar);
563 
564  if( ret != OM_SER_OK )
565  return(OM_SER_TIMEOUT);
566 
567  addr |= (unsigned int) thsChar;
568 
569  // is not destined for us?
570 
571  if( addr == OM_SER_BCAST_ADDR )
572  return(OM_SER_IS_BCAST);
573  else if( addr != m_devAddr )
574  return(OM_SER_NOT_US);
575 
576  } // end if(thsChar == 0
577  else {
578  // uint8_t invalid
579  return(OM_SER_ERR);
580  }
581 
582  return(OM_SER_OK);
583 
584 }

586 uint8_t OMMoCoBus::_getNextByte(uint8_t& p_data) {
587 
588  uint8_t ret = OM_SER_OK;
589  uint8_t* p = (uint8_t*)(void*) &p_data;
590 
591  if( m_serObj->available() > 0 ) {
592  *p = (uint8_t) m_serObj->read();
593  }
594  else {
595  unsigned long timer = millis();
596  // wait until we have data ready
597  while( m_serObj->available() < 1 ) {
598  if( millis() - timer > OM_SER_WAIT ) {
599  // timed out waiting for complete input
600  return( OM_SER_TIMEOUT );
601  }
602  }
603  *p = (uint8_t) m_serObj->read();
604  }
605 
606 
607  return(ret);
608 }
