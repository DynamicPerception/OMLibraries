221 uint8_t OMMenuMgr::checkInput() {
222 
223  static unsigned long lastTm = millis();
224  static uint8_t lastKey = BUTTON_NONE;
225  static uint8_t holdKey = BUTTON_NONE;
226  static unsigned int held = 0;
227 
228  // returning to list display after being
229  // disabled during an action
230  if( m_forceReturn == true && m_enable ) {
231  m_forceReturn = false;
232  _displayList(m_curParent, m_curTarget);
233  }
234 
235  // get which button is pressed
236  int key = BUTTON_NONE;
237 
238  if( m_inputType == MENU_ANALOG )
239  key = _checkAnalog();
240  else
241  key = _checkDigital();
242 
243 
244  if( key != lastKey ) {
245  // did not have two reads in the row with same key,
246  // thus de-bouncing is not possible
247  lastKey = key;
248  holdKey = BUTTON_NONE;
249  lastTm = millis();
250  held = 0;
251  m_holdMod = 1;
252  return BUTTON_NONE;
253  }
254 
255  // enough time for debounce?
256 
257  if( millis() - lastTm > OM_MENU_DEBOUNCE )
258  lastTm = millis();
259  else
260  return BUTTON_NONE;
261 
262  // no button is pressed, or the menu display
263  // is disabled
264  if( key == BUTTON_NONE || ! m_enable ) {
265  held = 0;
266  m_holdMod = 1;
267  holdKey = BUTTON_NONE;
268  return key;
269  }
270 
271  // if the menu hasn't been drawn, ignore
272  // interaction features
273 
274  if( key == BUTTON_SELECT )
275  m_menuActive = true;
276 
277  if( ! m_menuActive )
278  return key;
279 
280  // when a button is held, like up or down,
281  // we increase the modifier over time to allow
282  // the rate at which numbers go up/down to increase
283 
284  if( key == holdKey ) {
285 
286  held++;
287  if( held % 10 )
288  m_holdMod += 2;
289  }
290  else {
291  holdKey = key;
292  m_holdMod = 1;
293  }
294 
295 
296  _handleButton(key);
297 
298  return key;
299 
300 
301 }

381 int OMMenuMgr::_checkAnalog() {
382 
383  int curVal = analogRead(m_anaPin);
384 
385  static int lastVal = 0;
386 
387  // isolate jitter in a single cycle
388  if( abs(lastVal - curVal) > m_anaThresh ) {
389  lastVal = curVal;
390  return(BUTTON_NONE);
391  }
392 
393 
394  // check to see which analog button was read
395  for(byte i = 0; i < 5; i++ ) {
396  if( curVal > (m_butVals[i][0] - m_anaThresh) && curVal < (m_butVals[i][0] + m_anaThresh) )
397  return m_butVals[i][1];
398  }
399 
400  return BUTTON_NONE;
401 }

561 
562  // get item type
563  uint8_t type = pgm_read_byte(&(p_item->type));
564 
565  // process activation based on type
566  if( type == ITEM_VALUE ) {
567  m_inEdit = true;
568  _displayEdit(p_item);

578

589  if( m_enable )

642 void OMMenuMgr::_displayEdit(OMMenuItem* p_item) {
643 
644  // display label
645 
646  // copy data to buffer from progmem
647  memcpy_P(m_dispBuf, &( p_item->label ), sizeof(char) * sizeof(m_dispBuf) );
648 
649  _display(m_dispBuf, 0, 0, OM_MENU_COLS);
650 
651  OMMenuValue* value = reinterpret_cast<OMMenuValue*>(reinterpret_cast<void*>(pgm_read_word(&(p_item->target))));
652 
653  uint8_t type = pgm_read_byte(&(value->type));
654  void* valPtr = reinterpret_cast<void*>( pgm_read_word(&(value->targetValue)) );
655 
656  // get current value and squirrel it away for now
657  // we use temporary variables to avoid all sorts of crazyness should
658  // we want to port to another platform later, and endianness would
659  // be a problem if we decided to use one large variable (ulong) and
660  // then use specific bytes. This takes more memory, but it seems
661  // a viable trade
662 
663  if( type == TYPE_BYTE )
664  m_temp = *reinterpret_cast<uint8_t*>(valPtr);
665  else if( type == TYPE_UINT || type == TYPE_INT )
666  m_tempI = *reinterpret_cast<int*>(valPtr);
667  else if( type == TYPE_LONG || type == TYPE_ULONG )
668  m_tempL = *reinterpret_cast<long*>(valPtr);
669  else if( type == TYPE_SELECT ) {
670  // select types are interesting... We have a list of possible values and
671  // labels - we need to work that back to an index in the list...
672  OMMenuSelectValue* sel = reinterpret_cast<OMMenuSelectValue*>(valPtr);
673  OMMenuSelectListItem** list = reinterpret_cast<OMMenuSelectListItem**>(reinterpret_cast<void*>(pgm_read_word(&(sel->list))));
674  uint8_t curVal = *reinterpret_cast<uint8_t*>(pgm_read_word(&(sel->targetValue)));
675  uint8_t count = pgm_read_byte(&(sel->listCount));
676 
677  // mark to first index by default
678  m_temp = 0;
679 
680  // find index of current assigned value (if can be found)
681  for(uint8_t i = 0; i < count; i++ ) {
682  OMMenuSelectListItem* item = reinterpret_cast<OMMenuSelectListItem*>(pgm_read_word(&(list[i])));
683  uint8_t tgt = pgm_read_byte(&(item->value));
684  if( tgt == curVal )
685  m_temp = i;
686  }
687 
688  _displaySelVal(list, m_temp);
689  return;
690  }
691  else if( type == TYPE_BFLAG ) {
692  OMMenuValueFlag* flag = reinterpret_cast<OMMenuValueFlag*>(valPtr);
693  uint8_t* target = reinterpret_cast<uint8_t*>(pgm_read_word(&(flag->flag)));
694  uint8_t pos = pgm_read_byte(&(flag->pos));
695 
696  if( *target & (1 << pos) )
697  m_temp = 1;
698  else
699  m_temp = 0;
700 
701  _displayFlagVal();
702  return;
703  }
704  else if( type >= TYPE_FLOAT ) // always run as last check
705  m_tempF = *reinterpret_cast<float*>(valPtr);
706 
707  // throw number on-screen
708  _displayVoidNum(valPtr, type, 1, 0);
709 
710 
711 }

747 
748 void OMMenuMgr::_modifyTemp(uint8_t p_type, uint8_t p_mode, long p_min, long p_max) {
749 
750  void* tempNum;
751 
752  int mod = ( p_mode == MODE_INCREMENT ) ? 1 : -1;
753 
754  // apply holding rate mod
755  mod *= m_holdMod;
756 
757  // handle float precision adjustment
758  float fmod = (float) mod;
759 
760  if( p_type == TYPE_FLOAT_10 )
761  fmod /= 10.0;
762  else if( p_type == TYPE_FLOAT_100 )
763  fmod /= 100.0;
764  else if( p_type == TYPE_FLOAT_1000 )
765  fmod /= 1000.0;
766 
767 
768  // manage correct temporary variable change based on type
769  // and apply floor/ceiling from min/max
770 
771  if( p_type == TYPE_BYTE ) {
772  m_temp += mod;
773  if( p_min != 0 || p_max != 0 )
774  m_temp = m_temp > p_max ? p_max : ( m_temp < p_min ? p_min : m_temp );
775  tempNum = reinterpret_cast<void*>(&m_temp);
776  }
777  else if( p_type == TYPE_INT ) {
778  m_tempI += mod;
779  if( p_min != 0 || p_max != 0 )
780  m_tempI = m_tempI > p_max ? p_max : ( m_tempI < p_min ? p_min : m_tempI );
781  tempNum = reinterpret_cast<void*>(&m_tempI);
782  }
783  else if( p_type == TYPE_UINT ) {
784  *reinterpret_cast<unsigned int*>(&m_tempI) += mod;
785  if( p_min != 0 || p_max != 0 )
786  m_tempI = m_tempI > p_max ? p_max : ( m_tempI < p_min ? p_min : m_tempI );
787  tempNum = reinterpret_cast<void*>(&m_tempI);
788  }
789  else if( p_type == TYPE_LONG ) {
790  m_tempL += mod;
791  if( p_min != 0 || p_max != 0 )
792  m_tempL = m_tempL > p_max ? p_max : ( m_tempL < p_min ? p_min : m_tempL );
793  tempNum = reinterpret_cast<void*>(&m_tempL);
794  }
795  else if( p_type == TYPE_ULONG ) {
796  *reinterpret_cast<unsigned long*>(&m_tempL) += mod;
797  if( p_min != 0 || p_max != 0 )
798  m_tempL = m_tempL > p_max ? p_max : ( m_tempL < p_min ? p_min : m_tempL );
799  tempNum = reinterpret_cast<void*>(&m_tempL);
800  }
801  else if( p_type >= TYPE_FLOAT ) {
802  m_tempF += fmod;
803  if( p_min != 0 || p_max != 0 )
804  m_tempF = m_tempF > p_max ? p_max : ( m_tempF < p_min ? p_min : m_tempF );
805  tempNum = reinterpret_cast<void*>(&m_tempF);
806  }
807 
808  // display new temporary value
809  _displayVoidNum(tempNum, p_type, 1, 0);
810 
811 
812 }

882  OMMenuValue* thisValue = reinterpret_cast<OMMenuValue*>(pgm_read_word(&(p_item->target)));
883 
884  uint8_t type = pgm_read_byte(&(thisValue->type));
885  long min = pgm_read_dword(&(thisValue->min));
886  long max = pgm_read_dword(&(thisValue->max));
887 
888  uint8_t mode = (p_type == CHANGE_UP) ? MODE_INCREMENT : MODE_DECREMENT;
889 
890 
891  if( p_type == CHANGE_ABORT ) {
892  m_inEdit = false;
893  _activate(m_curParent, true);
894  }
895 
896  else if( p_type == CHANGE_UP || p_type == CHANGE_DOWN || p_type == MODE_NOOP ) {
897  if( type == TYPE_SELECT ) {
898  _modifySel(thisValue, mode);
899  }
900  else if( type == TYPE_BFLAG ) {
901  m_temp = (m_temp == 1) ? 0 : 1;
902  _displayFlagVal();
903  }
904  else
905  _modifyTemp(type, mode, min, max);
906  }
907 
908  else if( p_type == CHANGE_SAVE ) {
909 
910  void* ptr = reinterpret_cast<void*>(pgm_read_word(&(thisValue->targetValue)));
911 
912  if( type == TYPE_SELECT ) {
913  // select is more special than the rest, dig?
914 
915  // some what convoluted - we get the value stored in the current index (in m_temp) from the list,
916  // and store it in the byte pointer provided attached to the OMMenuSelectValue
917  OMMenuSelectValue* sel = reinterpret_cast<OMMenuSelectValue*>(ptr);
918  OMMenuSelectListItem** list = reinterpret_cast<OMMenuSelectListItem**>(reinterpret_cast<void*>(pgm_read_word(&(sel->list))));
919  OMMenuSelectListItem* item = reinterpret_cast<OMMenuSelectListItem*>(pgm_read_word(&(list[m_temp])));
920  uint8_t newVal = pgm_read_byte(&(item->value));
921  uint8_t* real = reinterpret_cast<uint8_t*>(pgm_read_word(&(sel->targetValue)));
922  *real = newVal;
923 
924  _eewrite<uint8_t>(thisValue, newVal);
925  }
926  else if( type == TYPE_BFLAG) {
927  // bflag is special too, we want to set a specific bit based on the current temp value
928  OMMenuValueFlag* flag = reinterpret_cast<OMMenuValueFlag*>(ptr);
929  uint8_t* target = reinterpret_cast<uint8_t*>(pgm_read_word(&(flag->flag)));
930  uint8_t pos = pgm_read_byte(&(flag->pos));
931 
932  if( m_temp )
933  *target |= (1 << pos);
934  else
935  *target &= (0xFF ^ (1 << pos));
936 
937  _eewrite<uint8_t>(thisValue, *target);
938  }
939  else if( type == TYPE_BYTE) {
940  *reinterpret_cast<uint8_t*>(ptr) = m_temp;
941  _eewrite<uint8_t>(thisValue, *reinterpret_cast<uint8_t*>(ptr));
942  }
943  else if( type == TYPE_UINT) {
944  *reinterpret_cast<unsigned int*>(ptr) = *reinterpret_cast<unsigned int*>(&m_tempI);
945  _eewrite<unsigned int>(thisValue, *reinterpret_cast<unsigned int*>(ptr));
946  }
947  else if( type == TYPE_INT ) {
948  *reinterpret_cast<int*>(ptr) = m_tempI;
949  _eewrite<int>(thisValue, *reinterpret_cast<int*>(ptr));
950  }
951  else if( type == TYPE_ULONG ) {
952  *reinterpret_cast<unsigned long*>(ptr) = *reinterpret_cast<unsigned long*>(&m_tempL);
953  _eewrite<unsigned long>(thisValue, *reinterpret_cast<unsigned long*>(ptr));
954  }
955  else if( type == TYPE_LONG ) {
956  *reinterpret_cast<long*>(ptr) = m_tempL;
957  _eewrite<long>(thisValue, *reinterpret_cast<long*>(ptr));
958  }
959  else if( type >= TYPE_FLOAT ) {
960  *reinterpret_cast<float*>(ptr) = m_tempF;
961  _eewrite<float>(thisValue, *reinterpret_cast<float*>(ptr));
962  }
963 
964  m_inEdit = false;
965  _activate(m_curParent, true);
966  }
967 }
