20 QList<QextPortInfo> QextSerialEnumerator::getPorts()
21 {
22  QList<QextPortInfo> infoList;
23 #ifdef Q_OS_LINUX
24  QStringList portNamePrefixes, portNameList;
25  portNamePrefixes << "ttyS*"; // list normal serial ports first
26 
27  QDir dir("/dev");
28  portNameList = dir.entryList(portNamePrefixes, (QDir::System | QDir::Files), QDir::Name);
29 
30  // remove the values which are not serial ports for e.g. /dev/ttysa
31  for (int i = 0; i < portNameList.size(); i++) {
32  bool ok;
33  QString current = portNameList.at(i);
34  // remove the ttyS part, and check, if the other part is a number
35  current.remove(0,4).toInt(&ok, 10);
36  if (!ok) {
37  portNameList.removeAt(i);
38  i--;
39  }
40  }
41 
42  // get the non standard serial ports names
43  // (USB-serial, bluetooth-serial, 18F PICs, and so on)
44  // if you know an other name prefix for serial ports please let us know
45  portNamePrefixes.clear();
46  portNamePrefixes << "ttyACM*" << "ttyUSB*" << "rfcomm*";
47  portNameList.append(dir.entryList(portNamePrefixes, (QDir::System | QDir::Files), QDir::Name));
48 
49  foreach (QString str , portNameList) {
50  QextPortInfo inf;
51  inf.physName = "/dev/"+str;
52  inf.portName = str;
53 
54  if (str.contains("ttyS")) {
55  inf.friendName = "Serial port "+str.remove(0, 4);
56  }
57  else if (str.contains("ttyUSB")) {
58  inf.friendName = "USB-serial adapter "+str.remove(0, 6);
59  }
60  else if (str.contains("rfcomm")) {
61  inf.friendName = "Bluetooth-serial adapter "+str.remove(0, 6);
62  }
63  inf.enumName = "/dev"; // is there a more helpful name for this?
64  infoList.append(inf);
65  }
66 #else
67  qCritical("Enumeration for POSIX systems (except Linux) is not implemented yet.");
68 #endif
69  return infoList;
70 }
