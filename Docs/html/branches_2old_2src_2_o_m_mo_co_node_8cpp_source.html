94 char OMMoCoNode::Poll(void) {
95  char eStatus = OM_NOERR;
96  eOMEventType eEvent;
97 
98  /* Check if the protocol stack is ready. */
99  if (eOMState != STATE_ENABLED) {
100  return OM_INVALID;
101  }
102 
103  /* Check if there is a event available. If not return control to caller.
104  * Otherwise we will handle the event. */
105  if (PortEventGet(&eEvent) == true) {
106  switch (eEvent) {
107  case OM_READY:
108  break;
109 
110  case OM_FRAME_RECEIVED:
111  eStatus = Receive(&ucRcvAddress, &ucOMFrame, &usLength);
112  if (eStatus == OM_NOERR) {
113  /* Check if the frame is for us. If not ignore the frame. */
114  if ((ucRcvAddress == ucMBAddress)) {
115  (void) PortEventPost(OM_EXECUTE);
116  }
117  }
118  break;
119 
120  case OM_EXECUTE:
121  ucFunctionCode = ucOMFrame[OM_PDU_FUNC_OFF];
122 
123  process(ucFunctionCode, &ucOMFrame, &usLength);
124 
125  eStatus = Send(ucMBAddress, ucOMFrame, usLength);
126 
127  break;
128 
129  case OM_FRAME_SENT:
130  break;
131  }
132  }
133  return OM_NOERR;
134 }
